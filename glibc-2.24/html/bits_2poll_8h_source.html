22 /* Event types that can be polled for. These bits may be set in `events'
23  to indicate the interesting event types; they will appear in `revents'
24  to indicate the status of the file descriptor. */
25 #define POLLIN 01 /* There is data to read. */
26 #define POLLPRI 02 /* There is urgent data to read. */
27 #define POLLOUT 04 /* Writing now will not block. */
28 
29 #if defined __USE_XOPEN || defined __USE_XOPEN2K8
30 /* These values are defined in XPG4.2 and later. */
31 # define POLLRDNORM POLLIN /* Normal data may be read. */
32 # define POLLRDBAND POLLPRI /* Priority data may be read. */
33 # define POLLWRNORM POLLOUT /* Writing now will not block. */
34 # define POLLWRBAND POLLOUT /* Priority data may be written. */
35 #endif
