25 /* There is always a big gap between the executable segment and the data
26  segments. Other code segments and data pages lie in there. So we must
27  unmap each segment individually (except for a codeless module). */
28 
29 static void __always_inline
30 _dl_unmap_segments (struct link_map *l)
31 {
32  if (l->l_contiguous)
33  /* Simple case. */
34  __munmap ((void *) l->l_map_start, l->l_map_end - l->l_map_start);
35  else
36  {
37  /* Normally l_phdr points into the RODATA segment, which we will
38  unmap in one iteration of the loop. So we cannot use it directly
39  throughout. */
40 
41  struct { ElfW(Addr) start, end; } segments[l->l_phnum], *seg = segments;
42 
43  for (const ElfW(Phdr) *ph = l->l_phdr; ph < &l->l_phdr[l->l_phnum]; ++ph)
44  if (ph->p_type == PT_LOAD)
45  {
46  seg->start = (l->l_addr + ph->p_vaddr) & -GLRO(dl_pagesize);
47  seg->end = (l->l_addr + ph->p_vaddr + ph->p_memsz
48  + GLRO(dl_pagesize) - 1) & -GLRO(dl_pagesize);
49  if (seg > segments && seg[-1].end == seg->start)
50  /* Coalesce two adjacent segments into one munmap call. */
51  seg[-1].end = seg->end;
52  else
53  ++seg;
54  }
55 
56  do
57  {
58  --seg;
59  __munmap ((void *) seg->start, seg->end - seg->start);
60  }
61  while (seg > segments);
62  }
63 }
