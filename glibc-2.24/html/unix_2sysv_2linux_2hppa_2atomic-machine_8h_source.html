64 /* The only basic operation needed is compare and exchange. The mem
65  pointer must be word aligned. */
66 #define atomic_compare_and_exchange_val_acq(mem, newval, oldval) \
67  ({ \
68  register long lws_errno asm("r21"); \
69  register unsigned long lws_ret asm("r28"); \
70  register unsigned long lws_mem asm("r26") = (unsigned long)(mem); \
71  register unsigned long lws_old asm("r25") = (unsigned long)(oldval);\
72  register unsigned long lws_new asm("r24") = (unsigned long)(newval);\
73  __asm__ __volatile__( \
74  "0: \n\t" \
75  "ble " _LWS "(%%sr2, %%r0) \n\t" \
76  "ldi " _LWS_CAS ", %%r20 \n\t" \
77  "ldi " _ASM_EAGAIN ", %%r20 \n\t" \
78  "cmpb,=,n %%r20, %%r21, 0b \n\t" \
79  "nop \n\t" \
80  "ldi " _ASM_EDEADLOCK ", %%r20 \n\t" \
81  "cmpb,=,n %%r20, %%r21, 0b \n\t" \
82  "nop \n\t" \
83  : "=r" (lws_ret), "=r" (lws_errno) \
84  : "r" (lws_mem), "r" (lws_old), "r" (lws_new) \
85  : _LWS_CLOBBER \
86  ); \
87  \
88  if (lws_errno == -EFAULT || lws_errno == -ENOSYS) \
89  ABORT_INSTRUCTION; \
90  \
91  (__typeof (oldval)) lws_ret; \
92  })
93
