40 #define DL_DT_FUNCTION_ADDRESS(map, start, attr, addr) \
41  attr volatile unsigned int fptr[2]; \
42  /* The test for "start & 2" below is to accommodate old binaries which \
43  violated the ELF ABI by pointing DT_INIT and DT_FINI at a function \
44  descriptor. */ \
45  if ((ElfW(Addr)) (start) & 2) \
46  addr = (ElfW(Addr)) start; \
47  else \
48  { \
49  fptr[0] = (unsigned int) (start); \
50  fptr[1] = (map)->l_info[DT_PLTGOT]->d_un.d_ptr; \
51  /* Set bit 30 to indicate to $$dyncall that this is a PLABEL. */ \
52  addr = (ElfW(Addr))((unsigned int)fptr | 2); \
53  } \
