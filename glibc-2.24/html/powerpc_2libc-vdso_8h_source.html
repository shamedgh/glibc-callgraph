43 /* The correct solution is for _dl_vdso_vsym to return the address of the OPD
44  for the kernel VDSO function. That address would then be stored in the
45  __vdso_* variables and returned as the result of the IFUNC resolver function.
46  Yet, the kernel does not contain any OPD entries for the VDSO functions
47  (incomplete implementation). However, PLT relocations for IFUNCs still expect
48  the address of an OPD to be returned from the IFUNC resolver function (since
49  PLT entries on PPC64 are just copies of OPDs). The solution for now is to
50  create an artificial static OPD for each VDSO function returned by a resolver
51  function. The TOC value is set to a non-zero value to avoid triggering lazy
52  symbol resolution via .glink0/.plt0 for a zero TOC (requires thread-safe PLT
53  sequences) when the dynamic linker isn't prepared for it e.g. RTLD_NOW. None
54  of the kernel VDSO routines use the TOC or AUX values so any non-zero value
55  will work. Note that function pointer comparisons will not use this artificial
56  static OPD since those are resolved via ADDR64 relocations and will point at
57  the non-IFUNC default OPD for the symbol. Lastly, because the IFUNC relocations
58  are processed immediately at startup the resolver functions and this code need
59  not be thread-safe, but if the caller writes to a PLT slot it must do so in a
60  thread-safe manner with all the required barriers. */
61 #define VDSO_IFUNC_RET(value) \
62  ({ \
63  static Elf64_FuncDesc vdso_opd = { .fd_toc = ~0x0 }; \
64  vdso_opd.fd_func = (Elf64_Addr)value; \
65  &vdso_opd; \
66  })
