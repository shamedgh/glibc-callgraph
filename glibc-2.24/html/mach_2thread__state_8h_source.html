47 static inline int
48 machine_get_state (thread_t thread, struct machine_thread_all_state *state,
49  int flavor, void *stateptr, void *scpptr, size_t size)
50 {
51  if (state->set & (1 << flavor))
52  {
53  /* Copy the saved state. */
54  memcpy (scpptr, stateptr, size);
55  return 1;
56  }
57  else
58  {
59  /* No one asked about this flavor of state before; fetch the state
60  directly from the kernel into the sigcontext. */
61  mach_msg_type_number_t got = (size / sizeof (int));
62  return (! __thread_get_state (thread, flavor, scpptr, &got)
63  && got == (size / sizeof (int)));
64  }
65 }
