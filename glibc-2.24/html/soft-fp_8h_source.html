260  do \

263  _FP_FRAC_ADDI_##wc (X, _FP_WORK_ROUND); \

286  do \

290  FP_SET_EXCEPTION (FP_EX_INEXACT); \
291  switch (FP_ROUNDMODE) \
292  { \
293  case FP_RND_NEAREST: \
