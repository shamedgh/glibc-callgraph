23 /* Look up the value of the next multibyte character and return its numerical
24  value if it is one of the digits known in the locale. If *DECIDED is
25  -1 this means it is not yet decided which form it is and we have to
26  search through all available digits. Otherwise we know which script
27  the digits are from. */
28 static inline int
29 indigit_value (const char **s, size_t *len, int *decided)
30 {
31  int from_level;
32  int to_level;
33  const char *mbdigits[10];
34  int i;
35  int n;
36 
37  if (*decided != -1)
38  from_level = to_level = *decided;
39  else
40  {
41  from_level = 0;
42  to_level = _NL_CURRENT_WORD (LC_CTYPE, _NL_CTYPE_INDIGITS_MB_LEN) - 1;
43  assert (from_level <= to_level);
44  }
45 
46  /* In this round we get the pointer to the digit strings and also perform
47  the first round of comparisons. */
48  for (n = 0; n < 10; ++n)
49  {
50  size_t dlen;
51 
52  /* Get the string for the digits with value N. */
53  mbdigits[n] = _NL_CURRENT (LC_CTYPE, _NL_CTYPE_INDIGITS0_MB + n);
54  dlen = strlen (mbdigits[n]);
55 
56  if (from_level == 0 && dlen <= *len
57  && memcmp (*s, mbdigits[n], dlen) == 0)
58  {
59  /* Found it. */
60  *s += dlen;
61  *len -= dlen;
62  if (*decided == -1)
63  *decided = 0;
64  return n;
65  }
66 
67  /* Advance the pointer to the next string. */
68  mbdigits[n] += dlen + 1;
69  }
70 
71  /* Now perform the remaining tests. */
72  for (i = 1; i <= to_level; ++i)
73  {
74  /* Search all ten digits of this level. */
75  for (n = 0; n < 10; ++n)
76  {
77  size_t dlen = strlen (mbdigits[n]);
78 
79  if (i >= from_level && dlen <= *len
80  && memcmp (*s, mbdigits[n], dlen) == 0)
81  {
82  /* Found it. */
83  *s += dlen;
84  *len -= dlen;
85  if (*decided == -1)
86  *decided = from_level;
87  return n;
88  }
89 
90  /* Advance the pointer to the next string. */
91  mbdigits[n] += dlen + 1;
92  }
93  }
94 
95  /* If we reach this point no matching digit was found. */
96  return -1;
97 }
