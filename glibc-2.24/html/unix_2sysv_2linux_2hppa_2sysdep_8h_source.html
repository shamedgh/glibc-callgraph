264 /* Linux takes system call arguments in registers:
265  syscall number gr20
266  arg 1 gr26
267  arg 2 gr25
268  arg 3 gr24
269  arg 4 gr23
270  arg 5 gr22
271  arg 6 gr21
272 
273  The compiler calls us by the C convention:
274  syscall number in the DO_CALL macro
275  arg 1 gr26
276  arg 2 gr25
277  arg 3 gr24
278  arg 4 gr23
279  arg 5 -52(sp)
280  arg 6 -56(sp)
281 
282  gr22 and gr21 are caller-saves, so we can just load the arguments
283  there and generally be happy. */

285 /* the cmpb...no_error code below inside DO_CALL
286  * is intended to mimic the if (__sys_res...)
287  * code inside INLINE_SYSCALL
288  */
289 #define NO_ERROR -0x1000
290 
291 #undef DO_CALL
292 #define DO_CALL(syscall_name, args) \
293  /* Create a frame */ ASM_LINE_SEP \
294  stwm TREG, 64(%sp) ASM_LINE_SEP \
295  .cfi_offset TREG, 0 ASM_LINE_SEP \
296  .cfi_adjust_cfa_offset 64 ASM_LINE_SEP \
297  stw %sp, -4(%sp) ASM_LINE_SEP \
298  .cfi_offset 30, -4 ASM_LINE_SEP \
299  stw %r19, -32(%sp) ASM_LINE_SEP \
300  .cfi_offset 19, -32 ASM_LINE_SEP \
301  /* Save r19 */ ASM_LINE_SEP \
302  SAVE_PIC(TREG) ASM_LINE_SEP \
303  /* Do syscall, delay loads # */ ASM_LINE_SEP \
304  ble 0x100(%sr2,%r0) ASM_LINE_SEP \
305  ldi SYS_ify (syscall_name), %r20 ASM_LINE_SEP \
306  ldi NO_ERROR,%r1 ASM_LINE_SEP \
307  cmpb,>>=,n %r1,%ret0,L(pre_end) ASM_LINE_SEP \
308  /* Restore r19 from TREG */ ASM_LINE_SEP \
309  LOAD_PIC(TREG) /* delay */ ASM_LINE_SEP \
310  SYSCALL_ERROR_HANDLER ASM_LINE_SEP \
311  /* Use TREG for temp storage */ ASM_LINE_SEP \
312  copy %ret0, TREG /* delay */ ASM_LINE_SEP \
313  /* OPTIMIZE: Don't reload r19 */ ASM_LINE_SEP \
314  /* do a -1*syscall_ret0 */ ASM_LINE_SEP \
315  sub %r0, TREG, TREG ASM_LINE_SEP \
316  /* Store into errno location */ ASM_LINE_SEP \
317  stw TREG, 0(%sr0,%ret0) ASM_LINE_SEP \
318  /* return -1 as error */ ASM_LINE_SEP \
319  ldo -1(%r0), %ret0 ASM_LINE_SEP \
320 L(pre_end): ASM_LINE_SEP \
321  /* Restore our frame, restoring TREG */ ASM_LINE_SEP \
322  ldwm -64(%sp), TREG ASM_LINE_SEP \
323  .cfi_adjust_cfa_offset -64 ASM_LINE_SEP \
324  /* Restore return pointer */ ASM_LINE_SEP \
325  ldw -20(%sp),%rp ASM_LINE_SEP \
326  .cfi_restore 2 ASM_LINE_SEP

354 /* GCC has to be warned that a syscall may clobber all the ABI
355  registers listed as "caller-saves", see page 8, Table 2
356  in section 2.2.6 of the PA-RISC RUN-TIME architecture
357  document. However! r28 is the result and will conflict with
358  the clobber list so it is left out. Also the input arguments
359  registers r20 -> r26 will conflict with the list so they
360  are treated specially. Although r19 is clobbered by the syscall
361  we cannot say this because it would violate ABI, thus we say
362  TREG is clobbered and use that register to save/restore r19
363  across the syscall. */
