62 # define _ISwbit(bit) (1 << (bit))
63 # else /* __BYTE_ORDER == __LITTLE_ENDIAN */
64 # define _ISwbit(bit) \
65  ((bit) < 8 ? (int) ((1UL << (bit)) << 24) \
66  : ((bit) < 16 ? (int) ((1UL << (bit)) << 8) \
