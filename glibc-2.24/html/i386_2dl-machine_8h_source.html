64 static inline int __attribute__ ((unused, always_inline))
65 elf_machine_runtime_setup (struct link_map *l, int lazy, int profile)
66 {
67  Elf32_Addr *got;
68  extern void _dl_runtime_resolve (Elf32_Word) attribute_hidden;
69  extern void _dl_runtime_profile (Elf32_Word) attribute_hidden;
70 
71  if (l->l_info[DT_JMPREL] && lazy)
72  {
73  /* The GOT entries for functions in the PLT have not yet been filled
74  in. Their initial contents will arrange when called to push an
75  offset into the .rel.plt section, push _GLOBAL_OFFSET_TABLE_[1],
76  and then jump to _GLOBAL_OFFSET_TABLE[2]. */
77  got = (Elf32_Addr *) D_PTR (l, l_info[DT_PLTGOT]);
78  /* If a library is prelinked but we have to relocate anyway,
79  we have to be able to undo the prelinking of .got.plt.
80  The prelinker saved us here address of .plt + 0x16. */
81  if (got[1])
82  {
83  l->l_mach.plt = got[1] + l->l_addr;
84  l->l_mach.gotplt = (Elf32_Addr) &got[3];
85  }
86  got[1] = (Elf32_Addr) l; /* Identify this shared object. */
87 
88  /* The got[2] entry contains the address of a function which gets
89  called to get the address of a so far unresolved function and
90  jump to it. The profiling extension of the dynamic linker allows
91  to intercept the calls to collect information. In this case we
92  don't store the address in the GOT so that all future calls also
93  end in this function. */
94  if (__glibc_unlikely (profile))
95  {
96  got[2] = (Elf32_Addr) &_dl_runtime_profile;
97 
98  if (GLRO(dl_profile) != NULL
99  && _dl_name_match_p (GLRO(dl_profile), l))
100  /* This is the object we are looking for. Say that we really
101  want profiling and the timers are started. */
102  GL(dl_profile_map) = l;
103  }
104  else
105  /* This function will get called to fix up the GOT entry indicated by
106  the offset on the stack, and then jump to the resolved address. */
107  got[2] = (Elf32_Addr) &_dl_runtime_resolve;
108  }
109 
110  return lazy;
111 }

140 /* Initial entry point code for the dynamic linker.
141  The C function `_dl_start' is the real entry point;
142  its return value is the user program's entry point. */
143 
144 #define RTLD_START asm ("\n\
145  .text\n\
146  .align 16\n\
147 0: movl (%esp), %ebx\n\
148  ret\n\
149  .align 16\n\
150 .globl _start\n\
151 .globl _dl_start_user\n\
152 _start:\n\
153  # Note that _dl_start gets the parameter in %eax.\n\
154  movl %esp, %eax\n\
155  call _dl_start\n\
156 _dl_start_user:\n\
157  # Save the user entry point address in %edi.\n\
158  movl %eax, %edi\n\
159  # Point %ebx at the GOT.\n\
160  call 0b\n\
161  addl $_GLOBAL_OFFSET_TABLE_, %ebx\n\
162  # See if we were run as a command with the executable file\n\
163  # name as an extra leading argument.\n\
164  movl _dl_skip_args@GOTOFF(%ebx), %eax\n\
165  # Pop the original argument count.\n\
166  popl %edx\n\
167  # Adjust the stack pointer to skip _dl_skip_args words.\n\
168  leal (%esp,%eax,4), %esp\n\
169  # Subtract _dl_skip_args from argc.\n\
170  subl %eax, %edx\n\
171  # Push argc back on the stack.\n\
172  push %edx\n\
173  # The special initializer gets called with the stack just\n\
174  # as the application's entry point will see it; it can\n\
175  # switch stacks if it moves these contents over.\n\
176 " RTLD_START_SPECIAL_INIT "\n\
177  # Load the parameters again.\n\
178  # (eax, edx, ecx, *--esp) = (_dl_loaded, argc, argv, envp)\n\
179  movl _rtld_local@GOTOFF(%ebx), %eax\n\
180  leal 8(%esp,%edx,4), %esi\n\
181  leal 4(%esp), %ecx\n\
182  movl %esp, %ebp\n\
183  # Make sure _dl_init is run with 16 byte aligned stack.\n\
184  andl $-16, %esp\n\
185  pushl %eax\n\
186  pushl %eax\n\
187  pushl %ebp\n\
188  pushl %esi\n\
189  # Clear %ebp, so that even constructors have terminated backchain.\n\
190  xorl %ebp, %ebp\n\
191  # Call the function to run the initializers.\n\
192  call _dl_init\n\
193  # Pass our finalizer function to the user in %edx, as per ELF ABI.\n\
194  leal _dl_fini@GOTOFF(%ebx), %edx\n\
195  # Restore %esp _start expects.\n\
196  movl (%esp), %esp\n\
197  # Jump to the user's entry point.\n\
198  jmp *%edi\n\
199  .previous\n\
200 ");

206 /* ELF_RTYPE_CLASS_PLT iff TYPE describes relocation of a PLT entry or
207  TLS variable, so undefined references should not be allowed to
208  define the value.
209  ELF_RTYPE_CLASS_COPY iff TYPE should not be allowed to resolve to one
210  of the main executable's symbols, as for a COPY reloc.
211  ELF_RTYPE_CLASS_EXTERN_PROTECTED_DATA iff TYPE describes relocation may
212  against protected data whose address be external due to copy relocation.
213  */

215  ((((type) == R_386_JMP_SLOT || (type) == R_386_TLS_DTPMOD32 \

217  || (type) == R_386_TLS_TPOFF || (type) == R_386_TLS_DESC) \
218  * ELF_RTYPE_CLASS_PLT) \
219  | (((type) == R_386_COPY) * ELF_RTYPE_CLASS_COPY) \
220  | (((type) == R_386_GLOB_DAT) * ELF_RTYPE_CLASS_EXTERN_PROTECTED_DATA))

280 auto inline void
281 __attribute ((always_inline))
282 elf_machine_rel (struct link_map *map, const Elf32_Rel *reloc,
283  const Elf32_Sym *sym, const struct r_found_version *version,
284  void *const reloc_addr_arg, int skip_ifunc)
285 {
286  Elf32_Addr *const reloc_addr = reloc_addr_arg;
287  const unsigned int r_type = ELF32_R_TYPE (reloc->r_info);
288 
289 # if !defined RTLD_BOOTSTRAP || !defined HAVE_Z_COMBRELOC
290  if (__glibc_unlikely (r_type == R_386_RELATIVE))
291  {
292 # if !defined RTLD_BOOTSTRAP && !defined HAVE_Z_COMBRELOC
293  /* This is defined in rtld.c, but nowhere in the static libc.a;
294  make the reference weak so static programs can still link.
295  This declaration cannot be done when compiling rtld.c
296  (i.e. #ifdef RTLD_BOOTSTRAP) because rtld.c contains the
297  common defn for _dl_rtld_map, which is incompatible with a
298  weak decl in the same file. */
299 # ifndef SHARED
300  weak_extern (_dl_rtld_map);
301 # endif
302  if (map != &GL(dl_rtld_map)) /* Already done in rtld itself. */
303 # endif
304  *reloc_addr += map->l_addr;
305  }
306 # ifndef RTLD_BOOTSTRAP
307  else if (__glibc_unlikely (r_type == R_386_NONE))
308  return;
309 # endif
310  else
311 # endif /* !RTLD_BOOTSTRAP and have no -z combreloc */
312  {
313 # ifndef RTLD_BOOTSTRAP
314  const Elf32_Sym *const refsym = sym;
315 # endif
316  struct link_map *sym_map = RESOLVE_MAP (&sym, version, r_type);
317  Elf32_Addr value = sym_map == NULL ? 0 : sym_map->l_addr + sym->st_value;
318 
319  if (sym != NULL
320  && __builtin_expect (ELFW(ST_TYPE) (sym->st_info) == STT_GNU_IFUNC,
321  0)
322  && __builtin_expect (sym->st_shndx != SHN_UNDEF, 1)
323  && __builtin_expect (!skip_ifunc, 1))
324  value = ((Elf32_Addr (*) (void)) value) ();
325 
326  switch (r_type)
327  {
328 # ifndef RTLD_BOOTSTRAP
329  case R_386_SIZE32:
330  /* Set to symbol size plus addend. */
331  *reloc_addr += sym->st_size;
332  break;
333 # endif
334  case R_386_GLOB_DAT:
335  case R_386_JMP_SLOT:
336  *reloc_addr = value;
337  break;
338 
339  case R_386_TLS_DTPMOD32:
340 # ifdef RTLD_BOOTSTRAP
341  /* During startup the dynamic linker is always the module
342  with index 1.
343  XXX If this relocation is necessary move before RESOLVE
344  call. */
345  *reloc_addr = 1;
346 # else
347  /* Get the information from the link map returned by the
348  resolv function. */
349  if (sym_map != NULL)
350  *reloc_addr = sym_map->l_tls_modid;
351 # endif
352  break;
353  case R_386_TLS_DTPOFF32:
354 # ifndef RTLD_BOOTSTRAP
355  /* During relocation all TLS symbols are defined and used.
356  Therefore the offset is already correct. */
357  if (sym != NULL)
358  *reloc_addr = sym->st_value;
359 # endif
360  break;
361  case R_386_TLS_DESC:
362  {
363  struct tlsdesc volatile *td =
364  (struct tlsdesc volatile *)reloc_addr;
365 
366 # ifndef RTLD_BOOTSTRAP
367  if (! sym)
368  td->entry = _dl_tlsdesc_undefweak;
369  else
370 # endif
371  {
372 # ifndef RTLD_BOOTSTRAP
373 # ifndef SHARED
374  CHECK_STATIC_TLS (map, sym_map);
375 # else
376  if (!TRY_STATIC_TLS (map, sym_map))
377  {
378  td->arg = _dl_make_tlsdesc_dynamic
379  (sym_map, sym->st_value + (ElfW(Word))td->arg);
380  td->entry = _dl_tlsdesc_dynamic;
381  }
382  else
383 # endif
384 # endif
385  {
386  td->arg = (void*)(sym->st_value - sym_map->l_tls_offset
387  + (ElfW(Word))td->arg);
388  td->entry = _dl_tlsdesc_return;
389  }
390  }
391  break;
392  }
393  case R_386_TLS_TPOFF32:
394  /* The offset is positive, backward from the thread pointer. */
395 # ifdef RTLD_BOOTSTRAP
396  *reloc_addr += map->l_tls_offset - sym->st_value;
397 # else
398  /* We know the offset of object the symbol is contained in.
399  It is a positive value which will be subtracted from the
400  thread pointer. To get the variable position in the TLS
401  block we subtract the offset from that of the TLS block. */
402  if (sym != NULL)
403  {
404  CHECK_STATIC_TLS (map, sym_map);
405  *reloc_addr += sym_map->l_tls_offset - sym->st_value;
406  }
407 # endif
408  break;
409  case R_386_TLS_TPOFF:
410  /* The offset is negative, forward from the thread pointer. */
411 # ifdef RTLD_BOOTSTRAP
412  *reloc_addr += sym->st_value - map->l_tls_offset;
413 # else
414  /* We know the offset of object the symbol is contained in.
415  It is a negative value which will be added to the
416  thread pointer. */
417  if (sym != NULL)
418  {
419  CHECK_STATIC_TLS (map, sym_map);
420  *reloc_addr += sym->st_value - sym_map->l_tls_offset;
421  }
422 # endif
423  break;
424 
425 # ifndef RTLD_BOOTSTRAP
426  case R_386_32:
427  *reloc_addr += value;
428  break;
429  case R_386_PC32:
430  *reloc_addr += (value - (Elf32_Addr) reloc_addr);
431  break;
432  case R_386_COPY:
433  if (sym == NULL)
434  /* This can happen in trace mode if an object could not be
435  found. */
436  break;
437  if (__builtin_expect (sym->st_size > refsym->st_size, 0)
438  || (__builtin_expect (sym->st_size < refsym->st_size, 0)
439  && GLRO(dl_verbose)))
440  {
441  const char *strtab;
442 
443  strtab = (const char *) D_PTR (map, l_info[DT_STRTAB]);
444  _dl_error_printf ("\
445 %s: Symbol `%s' has different size in shared object, consider re-linking\n",
446  RTLD_PROGNAME, strtab + refsym->st_name);
447  }
448  memcpy (reloc_addr_arg, (void *) value,
449  MIN (sym->st_size, refsym->st_size));
450  break;
451  case R_386_IRELATIVE:
452  value = map->l_addr + *reloc_addr;
453  value = ((Elf32_Addr (*) (void)) value) ();
454  *reloc_addr = value;
455  break;
456  default:
457  _dl_reloc_bad_type (map, r_type, 0);
458  break;
459 # endif /* !RTLD_BOOTSTRAP */
460  }
461  }
462 }

466 __attribute__ ((always_inline))
467 elf_machine_rela (struct link_map *map, const Elf32_Rela *reloc,
468  const Elf32_Sym *sym, const struct r_found_version *version,
469  void *const reloc_addr_arg, int skip_ifunc)
470 {
471  Elf32_Addr *const reloc_addr = reloc_addr_arg;
472  const unsigned int r_type = ELF32_R_TYPE (reloc->r_info);
473 
474  if (ELF32_R_TYPE (reloc->r_info) == R_386_RELATIVE)
475  *reloc_addr = map->l_addr + reloc->r_addend;
476  else if (r_type != R_386_NONE)
477  {
478 # ifndef RESOLVE_CONFLICT_FIND_MAP
479  const Elf32_Sym *const refsym = sym;
480 # endif
481  struct link_map *sym_map = RESOLVE_MAP (&sym, version, r_type);
482  Elf32_Addr value = sym == NULL ? 0 : sym_map->l_addr + sym->st_value;
483 
484  if (sym != NULL
485  && __builtin_expect (sym->st_shndx != SHN_UNDEF, 1)
486  && __builtin_expect (ELFW(ST_TYPE) (sym->st_info) == STT_GNU_IFUNC, 0)
487  && __builtin_expect (!skip_ifunc, 1))
488  value = ((Elf32_Addr (*) (void)) value) ();
489 
490  switch (ELF32_R_TYPE (reloc->r_info))
491  {
492  case R_386_SIZE32:
493  /* Set to symbol size plus addend. */
494  value = sym->st_size;
495  case R_386_GLOB_DAT:
496  case R_386_JMP_SLOT:
497  case R_386_32:
498  *reloc_addr = value + reloc->r_addend;
499  break;
500 # ifndef RESOLVE_CONFLICT_FIND_MAP
501  /* Not needed for dl-conflict.c. */
502  case R_386_PC32:
503  *reloc_addr = (value + reloc->r_addend - (Elf32_Addr) reloc_addr);
504  break;
505 
506  case R_386_TLS_DTPMOD32:
507  /* Get the information from the link map returned by the
508  resolv function. */
509  if (sym_map != NULL)
510  *reloc_addr = sym_map->l_tls_modid;
511  break;
512  case R_386_TLS_DTPOFF32:
513  /* During relocation all TLS symbols are defined and used.
514  Therefore the offset is already correct. */
515  *reloc_addr = (sym == NULL ? 0 : sym->st_value) + reloc->r_addend;
516  break;
517  case R_386_TLS_DESC:
518  {
519  struct tlsdesc volatile *td =
520  (struct tlsdesc volatile *)reloc_addr;
521 
522 # ifndef RTLD_BOOTSTRAP
523  if (!sym)
524  {
525  td->arg = (void*)reloc->r_addend;
526  td->entry = _dl_tlsdesc_undefweak;
527  }
528  else
529 # endif
530  {
531 # ifndef RTLD_BOOTSTRAP
532 # ifndef SHARED
533  CHECK_STATIC_TLS (map, sym_map);
534 # else
535  if (!TRY_STATIC_TLS (map, sym_map))
536  {
537  td->arg = _dl_make_tlsdesc_dynamic
538  (sym_map, sym->st_value + reloc->r_addend);
539  td->entry = _dl_tlsdesc_dynamic;
540  }
541  else
542 # endif
543 # endif
544  {
545  td->arg = (void*)(sym->st_value - sym_map->l_tls_offset
546  + reloc->r_addend);
547  td->entry = _dl_tlsdesc_return;
548  }
549  }
550  }
551  break;
552  case R_386_TLS_TPOFF32:
553  /* The offset is positive, backward from the thread pointer. */
554  /* We know the offset of object the symbol is contained in.
555  It is a positive value which will be subtracted from the
556  thread pointer. To get the variable position in the TLS
557  block we subtract the offset from that of the TLS block. */
558  if (sym != NULL)
559  {
560  CHECK_STATIC_TLS (map, sym_map);
561  *reloc_addr = sym_map->l_tls_offset - sym->st_value
562  + reloc->r_addend;
563  }
564  break;
565  case R_386_TLS_TPOFF:
566  /* The offset is negative, forward from the thread pointer. */
567  /* We know the offset of object the symbol is contained in.
568  It is a negative value which will be added to the
569  thread pointer. */
570  if (sym != NULL)
571  {
572  CHECK_STATIC_TLS (map, sym_map);
573  *reloc_addr = sym->st_value - sym_map->l_tls_offset
574  + reloc->r_addend;
575  }
576  break;
577  case R_386_COPY:
578  if (sym == NULL)
579  /* This can happen in trace mode if an object could not be
580  found. */
581  break;
582  if (__builtin_expect (sym->st_size > refsym->st_size, 0)
583  || (__builtin_expect (sym->st_size < refsym->st_size, 0)
584  && GLRO(dl_verbose)))
585  {
586  const char *strtab;
587 
588  strtab = (const char *) D_PTR (map, l_info[DT_STRTAB]);
589  _dl_error_printf ("\
590 %s: Symbol `%s' has different size in shared object, consider re-linking\n",
591  RTLD_PROGNAME, strtab + refsym->st_name);
592  }
593  memcpy (reloc_addr_arg, (void *) value,
594  MIN (sym->st_size, refsym->st_size));
595  break;
