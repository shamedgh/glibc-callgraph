41 /* Wait until a lll_futex_wake call on FUTEXP, or TIMEOUT elapses. */
42 #define lll_futex_timed_wait(futexp, val, timeout, private) \
43  ({ \
44  /* This timeout is relative, but the IRT call wants it absolute. */\
45  const struct timespec *_to = (timeout); \
46  struct timespec _ts; \
47  int _err = 0; \
48  if (_to != NULL \
49  && __glibc_likely ((_err = __nacl_irt_clock.clock_gettime \
50  (CLOCK_REALTIME, &_ts)) == 0)) \
51  { \
52  _ts.tv_sec += _to->tv_sec; \
53  _ts.tv_nsec += _to->tv_nsec; \
54  while (_ts.tv_nsec >= 1000000000) \
55  { \
56  _ts.tv_nsec -= 1000000000; \
57  ++_ts.tv_sec; \
58  } \
59  _to = &_ts; \
60  } \
61  if (_err == 0) \
62  _err = __nacl_irt_futex.futex_wait_abs \
63  ((volatile int *) (futexp), val, _to); \
64  (void) (private); \
65  -_err; \
66  })
