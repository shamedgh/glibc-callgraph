67 #ifndef DONT_NEED_READ_INT
68 /* Read a simple integer from a string and update the string pointer.
69  It is assumed that the first character is a digit. */
70 static int
71 read_int (const UCHAR_T * *pstr)
72 {
73  int retval = **pstr - L_('0');
74 
75  while (ISDIGIT (*++(*pstr)))
76  if (retval >= 0)
77  {
78  if (INT_MAX / 10 < retval)
79  retval = -1;
80  else
81  {
82  int digit = **pstr - L_('0');
83 
84  retval *= 10;
85  if (INT_MAX - digit < retval)
86  retval = -1;
87  else
88  retval += digit;
89  }
90  }
91 
92  return retval;
93 }
