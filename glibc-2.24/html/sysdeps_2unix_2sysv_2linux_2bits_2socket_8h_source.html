295 extern struct cmsghdr *__cmsg_nxthdr (struct msghdr *__mhdr,
296  struct cmsghdr *__cmsg) __THROW;
297 #ifdef __USE_EXTERN_INLINES

301 _EXTERN_INLINE struct cmsghdr *
302 __NTH (__cmsg_nxthdr (struct msghdr *__mhdr, struct cmsghdr *__cmsg))
303 {
304  if ((size_t) __cmsg->cmsg_len < sizeof (struct cmsghdr))
305  /* The kernel header does this so there may be a reason. */
306  return (struct cmsghdr *) 0;
307 
308  __cmsg = (struct cmsghdr *) ((unsigned char *) __cmsg
309  + CMSG_ALIGN (__cmsg->cmsg_len));
310  if ((unsigned char *) (__cmsg + 1) > ((unsigned char *) __mhdr->msg_control
311  + __mhdr->msg_controllen)
312  || ((unsigned char *) __cmsg + CMSG_ALIGN (__cmsg->cmsg_len)
313  > ((unsigned char *) __mhdr->msg_control + __mhdr->msg_controllen)))
314  /* No more entries. */
315  return (struct cmsghdr *) 0;
