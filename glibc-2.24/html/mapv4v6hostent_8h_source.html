62 {

65  if (hp->h_addrtype != AF_INET || hp->h_length != INADDRSZ)
66  return 0;
67  hp->h_addrtype = AF_INET6;
68  hp->h_length = IN6ADDRSZ;

72 
73  if (*lenp < (i + IN6ADDRSZ))
74  /* Out of memory. */
75  return 1;

78  map_v4v6_address (*ap, *bpp);

81  *lenp -= IN6ADDRSZ;
82  }
83  return 0;
