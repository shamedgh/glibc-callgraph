87 #define ICMP6_FILTER_WILLPASS(type, filterp) \
88  ((((filterp)->icmp6_filt[(type) >> 5]) & (1 << ((type) & 31))) == 0)

90 #define ICMP6_FILTER_WILLBLOCK(type, filterp) \
