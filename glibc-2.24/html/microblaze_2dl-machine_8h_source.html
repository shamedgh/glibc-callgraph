34 /* Return the link-time address of _DYNAMIC. Conveniently, this is the
35  first element of the GOT. This must be inlined in a function which
36  uses global data. */
37 static inline Elf32_Addr
38 elf_machine_dynamic (void)
39 {
40  /* This produces a GOTOFF reloc that resolves to zero at link time, so in
41  fact just loads from the GOT register directly. By doing it without
42  an asm we can let the compiler choose any register. */
43 
44  Elf32_Addr got_entry_0;
45  __asm__ __volatile__(
46  "lwi %0,r20,0"
47  :"=r"(got_entry_0)
48  );
49  return got_entry_0;
50 }

52 /* Return the run-time load address of the shared object. */
53 static inline Elf32_Addr
54 elf_machine_load_address (void)
55 {
56  /* Compute the difference between the runtime address of _DYNAMIC as seen
57  by a GOTOFF reference, and the link-time address found in the special
58  unrelocated first GOT entry. */
59 
60  Elf32_Addr dyn;
61  __asm__ __volatile__ (
62  "addik %0,r20,_DYNAMIC@GOTOFF"
63  : "=r"(dyn)
64  );
65  return dyn - elf_machine_dynamic ();
66 }

149 /* ELF_RTYPE_CLASS_PLT iff TYPE describes relocation of a PLT entry or
150  TLS variable, so undefined references should not be allowed to
151  define the value.
152  ELF_RTYPE_CLASS_COPY iff TYPE should not be allowed to resolve to one
153  of the main executable's symbols, as for a COPY reloc. */
154 #ifndef RTLD_BOOTSTRAP

159  (type) == R_MICROBLAZE_TLSTPREL32) \

209 auto inline void __attribute__ ((always_inline))
210 elf_machine_rela (struct link_map *map, const Elf32_Rela *reloc,
211  const Elf32_Sym *sym, const struct r_found_version *version,
212  void *const reloc_addr_arg, int skip_ifunc)
213 {
214  Elf32_Addr *const reloc_addr = reloc_addr_arg;
215  const int r_type = ELF32_R_TYPE (reloc->r_info);
216 
217  if (__builtin_expect (r_type == R_MICROBLAZE_64_PCREL, 0))
218  PUT_REL_64 (reloc_addr, map->l_addr + reloc->r_addend);
219  else if (r_type == R_MICROBLAZE_REL)
220  *reloc_addr = map->l_addr + reloc->r_addend;
221  else
222  {
223  const Elf32_Sym *const refsym = sym;
224  struct link_map *sym_map = RESOLVE_MAP (&sym, version, r_type);
225  Elf32_Addr value = sym == NULL ? 0 : sym_map->l_addr + sym->st_value;
226 
227  value += reloc->r_addend;
228  if (r_type == R_MICROBLAZE_GLOB_DAT ||
229  r_type == R_MICROBLAZE_JUMP_SLOT ||
230  r_type == R_MICROBLAZE_32)
231  {
232  *reloc_addr = value;
233  }
234  else if (r_type == R_MICROBLAZE_COPY)
235  {
236  if (sym != NULL && (sym->st_size > refsym->st_size
237  || (sym->st_size < refsym->st_size && GLRO (dl_verbose))) )
238  {
239  const char *strtab;
240 
241  strtab = (const void *) D_PTR (map, l_info[DT_STRTAB]);
242  _dl_error_printf ("\
243 %s: Symbol `%s' has different size in shared object, consider re-linking\n",
244  RTLD_PROGNAME, strtab + refsym->st_name);
245  }
246  memcpy (reloc_addr_arg, (void *) value,
247  MIN (sym->st_size, refsym->st_size));
248  }
249  else if (r_type == R_MICROBLAZE_NONE)
250  {
251  }
252 #if !defined RTLD_BOOTSTRAP
253  else if (r_type == R_MICROBLAZE_TLSDTPMOD32)
254  {
255  if (sym_map != NULL)
256  *reloc_addr = sym_map->l_tls_modid;
257  }
258  else if (r_type == R_MICROBLAZE_TLSDTPREL32)
259  {
260  if (sym != NULL)
261  *reloc_addr = sym->st_value + reloc->r_addend;
262  }
263  else if (r_type == R_MICROBLAZE_TLSTPREL32)
264  {
265  if (sym != NULL)
266  {
267  CHECK_STATIC_TLS (map, sym_map);
268  *reloc_addr = sym->st_value + sym_map->l_tls_offset + reloc->r_addend;
269  }
270  }
271 #endif
272  else
273  {
274  _dl_reloc_bad_type (map, r_type, 0);
275  }
276  }
277 }
